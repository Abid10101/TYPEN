<body>
    <h1>NEET Typing Practice</h1>
    <input type="file" id="pdfUpload" accept="application/pdf">
    <button onclick="extractText()">Extract Key Points</button>
    <p id="sentence">Upload a PDF and click "Extract Key Points".</p>
    <textarea id="typingArea" placeholder="Start typing here..." disabled></textarea>
    <br><br>
    <button onclick="startTest()">Start</button>
    <p id="result"></p>
    
    <script>
        let keyPoints = [];
        let currentIndex = 0;
        let startTime;

        function extractText() {
            const file = document.getElementById("pdfUpload").files[0];
            if (!file) {
                alert("Please upload a PDF first.");
                return;
            }
            
            const reader = new FileReader();
            reader.onload = function() {
                const pdfText = reader.result;
                keyPoints = generateKeyPoints(pdfText);
                if (keyPoints.length > 0) {
                    document.getElementById("sentence").innerText = "Key points extracted. Click Start to practice.";
                } else {
                    alert("No key points found. Try another file.");
                }
            };
            reader.readAsText(file);
        }
        
        function generateKeyPoints(text) {
            let lines = text.split(". ");
            return lines.filter(line => line.length > 20); 
        }
        
        function startTest() {
            if (keyPoints.length === 0) {
                alert("Please extract key points first.");
                return;
            }
            currentIndex = 0;
            displayNextSentence();
        }
        
        function displayNextSentence() {
            if (currentIndex < keyPoints.length) {
                document.getElementById("sentence").innerText = keyPoints[currentIndex];
                document.getElementById("typingArea").value = "";
                document.getElementById("typingArea").disabled = false;
                document.getElementById("typingArea").focus();
                startTime = new Date().getTime();
            } else {
                document.getElementById("sentence").innerText = "All key points completed!";
                document.getElementById("typingArea").disabled = true;
            }
        }
        
        document.getElementById("typingArea").addEventListener("input", function() {
            let typedText = this.value;
            if (typedText === keyPoints[currentIndex]) {
                let endTime = new Date().getTime();
                let timeTaken = (endTime - startTime) / 1000;
                let words = keyPoints[currentIndex].split(" ").length;
                let speed = (words / (timeTaken / 60)).toFixed(2);
                document.getElementById("result").innerText = `Completed in ${timeTaken} seconds! Speed: ${speed} WPM`;
                currentIndex++;
                setTimeout(displayNextSentence, 1000);
            }
        });
    </script>
